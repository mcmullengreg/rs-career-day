  1. Greg McMullen

    1. What is a Web Developer

      A programmer specializing in the development of World Wide Web applications
      that are run over HTTP from a web server to a browser [1]

      Web Dev's can focus on front-end (user interaction) or
      back-end (server side frameworks and database management)

    2. A day in the life

    3. I can make it snow…

      								function snow(){
								    snctx.clearRect(0, 0, sn.width, sn.height);

								    for (var i = 0; i < flakeCount; i++) {
								        var flake = flakes[i],
								            x = mX,
								            y = mY,
								            minDist = 150,
								            x2 = flake.x,
								            y2 = flake.y;
								
								        var dist = Math.sqrt((x2 - x) * (x2 - x) + (y2 - y) * (y2 - y)),
								            dx = x2 - x,
								            dy = y2 - y;
								
								        if (dist < minDist) {
								            var force = minDist / (dist * dist),
								                xcomp = (x - x2) / dist,
								                ycomp = (y - y2) / dist,
								                deltaV = force / 2;
								
								            flake.velX -= deltaV * xcomp;
								            flake.velY -= deltaV * ycomp;
								
								        }
								        snctx.fillStyle = "rgba(255,255,255," + flake.opacity + ")";
								        flake.y += flake.velY;
								        flake.x += flake.velX;    
								        if (flake.y >= sn.height || flake.y <= 0) {
								            reset(flake);
								        }
										snctx.beginPath();
								        snctx.arc(flake.x, flake.y, flake.size, 0, Math.PI * 2);
								        snctx.fill();
								    }
								    requestAnimationFrame(snow);
								}

									this.coordinates = [];
									this.coordinateCount = 3;

									while( this.coordinateCount-- ) {

									}
									this.angle = Math.atan2( ty - sy, tx - sx );
